Key implementation notes:

1. **isAvailable()**: `return require('fs').existsSync('/usr/bin/journalctl');` — synchronous, cheap, accurate for native server.

2. **listContainers()**: List linbo-* systemd units:
```javascript
async function listContainers() {
  return new Promise((resolve) => {
    execFile('/usr/bin/journalctl', ['--field=_SYSTEMD_UNIT', '--no-pager'], (err, stdout) => {
      if (err || !stdout) return resolve([]);
      const units = [...new Set(
        stdout.trim().split('\n')
          .map(u => u.trim())
          .filter(u => u.startsWith('linbo-') && u.endsWith('.service'))
      )];
      resolve(units.map(unit => ({
        name: unit.replace(/\.service$/, ''),
        id: 'systemd',
        state: 'active',
        status: 'running (journald)',
        image: '',
      })));
    });
  });
}
```
